


// STATEFUL COMPONENTS

class MyComponent extends React.Component {
  constructor(props) {
    super(props);
    this.state = {
      name: 'freeCodeCamp'
    }
  }
  render() {
    const name = this.state.name;
    return (
      <div>
        <h1>{name}</h1>
      </div>
    );
  }
};


// STATEFUL COMPONENTS

class StatefulComponent extends React.Component {
  constructor(props) {

    super(props);

    this.state = {firstName : "name"}

  }
  render() {
    return (
      <div>
        <h1>{this.state.firstName}</h1>
      </div>
    );
  }
};
